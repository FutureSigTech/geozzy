    Get chart type from a metric and a group ID

    exports.getChartType = function(metric,groupBy){
    if (metric.type == constants.METRIC_TYPE_RESOURCE){
        var barChartGroups = [
            constants.GROUP_BY_RESOURCE_NAME,
            constants.GROUP_BY_RESOURCE_TYPE,
            constants.GROUP_BY_RESOURCE_TOPIC,
            constants.GROUP_BY_RESOURCE_TERM,
            constants.GROUP_BY_DEVICE_TYPE,
            constants.GROUP_BY_LANGUAGE,
            constants.GROUP_BY_SECTION
        ];
        if (_.contains(barChartGroups,groupBy.groupByID)){
            return constants.BAR_CHART;
        }
        if (groupBy.groupByID == constants.GROUP_BY_GEOGRAPHIC_BOUND){
            return constants.HEAT_MAP;
        }
        if (groupBy.groupByID == constants.GROUP_BY_DATE_RANGE){
            return constants.DATE_CHART;
        }
        if (groupBy.groupByID == constants.GROUP_BY_CLIENT_LOCATION){
            return constants.COUNTRY_BAR_CHART;
        }
    }
    if (metric.type == constants.METRIC_TYPE_EXPLORER){
        if (metric.metricID == constants.METRIC_EXPLORED_ZONE){
            if (groupBy.groupByID == constants.GROUP_BY_EXPLORER_BOUNDS ||
                groupBy.groupByID== constants.GROUP_BY_FILTER_BOUNDS ){
                return constants.MAP_EXPLORED_ZONES;
            }
        }
        if (metric.metricID == constants.METRIC_FILTER){
            var barChartGroups = [
                constants.GROUP_BY_EXPLORER,
                constants.GROUP_BY_DEVICE_TYPE,
                constants.GROUP_BY_LANGUAGE
            ];
            if (_.contains(barChartGroups,groupBy.groupByID)){
                return constants.BAR_CHART;
            }
            if (groupBy.groupByID == constants.GROUP_BY_DATE_RANGE){
                return constants.DATE_CHART;
            }
            if (groupBy.groupByID == constants.GROUP_BY_CLIENT_LOCATION){
                return constants.COUNTRY_BAR_CHART;
            }
        }
    }
    return undefined;
};

    Get filters related to a metric type

    exports.getFilters = function(metric){
    var filters = [],
        filterIDs = [];
    if (metric.type == constants.METRIC_TYPE_RESOURCE){
        filterIDs = [
            constants.FILTER_RESOURCE,
            constants.FILTER_RESOURCE_TYPES,
            constants.FILTER_RESOURCE_TOPICS,
            constants.FILTER_RESOURCE_TERMS,
            constants.FILTER_EVENT_TYPES,
            constants.FILTER_REGION_COUNTRY,
            constants.FILTER_DEVICE_TYPE,
            constants.FILTER_LANGUAGE,
            constants.FILTER_DATE_RANGE,
            constants.FILTER_RESOURCE_GEOGRAPHIC_BOUND,
            constants.FILTER_SECTION,
            constants.FILTER_SECONDS
        ];
    }
    if (metric.type == constants.METRIC_TYPE_EXPLORER){
        filterIDs = [
            constants.FILTER_EXPLORER,
            constants.FILTER_TAXONOMY_TERMS,
            constants.FILTER_EXPLORER_GEOGRAPHIC_BOUND,
            constants.FILTER_DATE_RANGE,
            constants.FILTER_REGION_COUNTRY,
            constants.FILTER_DEVICE_TYPE,
            constants.FILTER_SECTION
        ];
    }
    _.each(filterIDs, function(filterID){
        var filter = _.findWhere(configFilters,{filterID:filterID});
        if (!_.isUndefined(filter)){
            var pickedFilter = _.pick(filter,'filterID','filterType','filterName');
            filters.push(pickedFilter);
        }
    });
    return filters;
};

    Get some $match MongoDB pipeline stages using the request metric and the request filters

    exports.getRequestFiltersMatch = function(metric,requestFilters){
    var filters = [];
    _.each(requestFilters,function(requestFilter){
        var configFilter = _.findWhere(configFilters,{filterID:requestFilter.filterID});
        if (!_.isUndefined(configFilter) && !_.isEmpty(requestFilter.values)){
            var field = configFilter.field;
            var operator = configFilter.operator;
            _.each(requestFilter.values, function(requestFilter){
                operator = !_.isUndefined(requestFilter.operator) ? requestFilter.operator : operator;
                var value = !_.isUndefined(requestFilter.value) ? requestFilter.value : requestFilter;

                if (configFilter.isInteger) {
                    value = parseInt(value);
                }
                if (configFilter.isDate){
                    value = new Date(value);
                }

                var filter = {};
                filter[field] = {};
                filter[field][operator] = value;
                filters.push(filter);
            });
        }
    });
    if (!_.isEmpty(filters)){
        return {
            $and: filters
        }
    }else{
        return undefined;
    }
};

    exports.getMatchFilter = function(filterID,paramValue){
    var configFilter = _.findWhere(configFilters, {filterID: filterID});
    if (!_.isUndefined(configFilter)){
        var field = configFilter.field;
        var operator = configFilter.operator;
        var configValue = configFilter.value;

        var filter = {};
        filter[field] = {};
        filter[field][operator] = !_.isUndefined(configValue)? configValue:paramValue;
    }
    return filter;
};
