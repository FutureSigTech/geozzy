    Check array type and not undefined

    var validArray = function(array){
    return (!_.isUndefined(array) && (array instanceof Array));
};

    Check order param is valid

    var validOrder = function(order){
    return (!_.isUndefined(order) && (order == 1 || order == -1));
};

exports.validArray = validArray;
exports.validOrder = validOrder;

    Check required stats params

    exports.validStatisticParams = function(metricID,groupByID,filters,order){
    if (_.isUndefined(metricID) ||
        _.isUndefined(groupByID)){
        return false;
    }
    groupByID = parseInt(groupByID);
    return !(_.isUndefined(groupByID) ||
        (!_.isUndefined(filters) && !validArray(filters)) ||
        (!_.isUndefined(order) && !validOrder(order)));
};
