    Configuration of a single thread NodeJS server

    Dependency configuration:

    • express: NodeJS server library,
    • app: Express app configured with middlewares and routes
    • http: NodeJS http library
    • server: NodeJS server instance
    • io: Socket.io instance of the current server
    • socket: Module of socket functions
    • mongoose: MongoDB object moddelling for NodeJS
    • configDB: Database module configuration
    • configServer: Server module configuration
    • server_ip_addrees: It will be setted up to a enviroment variable if exists, else it will have a server configuration file value
    • server_port: Same as server_ip_address
    var express = require('./server/config/express/express'),
    app = express.createApp(),
    http = require('http'),
    server = http.createServer(app),
    io = require('socket.io').listen(server),
    socket = require('./server/sockets/socket'),
    mongoose = require('mongoose'),
    configDB = require('./server/config/system/configDB'),
    configServer = require('./server/config/system/configServer'),
    server_ip_address = process.env.NODEJS_IP || configServer.IP,
    server_port = process.env.NODEJS_PORT || configServer.port;
