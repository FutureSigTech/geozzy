    Get all metrics

    exports.getMetrics = function(req,res){
    var metrics = [];
    _.each(configMetrics,function(metric){
        if (!_.isUndefined(metric.metricName) && !_.isEmpty(metric.metricName)){
            metrics.push( _.pick(metric,"metricID","metricName","type"));
        }
    });
    if (_.isUndefined(metrics) || _.size(metrics) === 0){
        return res.status(404);
    }
    return res.json(metrics);
};

    Get organizations of a metric

    exports.getOrganizations = function(req,res){
    var metricID = req.query.metricID;
    if (_.isUndefined(metricID)){
        return res.send(400);
    }
    var groupsIDs = configUtils.getMetricGroupIDs(metricID);
    var organizations =
        _.map(
            _.map(
                groupsIDs,
                function(groupID){
                    return configUtils.getGroupBy(groupID);
                }
            ),
            function(groupBy){
                return _.pick(groupBy,"groupByID","groupByName");
            }
        );
    if (_.isUndefined(organizations) || _.size(organizations) === 0){
        return res.status(404);
    }
    return res.json(organizations);
};

    Get filters of a metric

    exports.getFilters = function(req,res){
    var metricID = req.query.metricID;
    if (_.isUndefined(metricID)){
        return res.send(404);
    }
    var metric = configUtils.getMetric(metricID);
    if (_.isUndefined(metric)){
        return res.send(404);
    }
    var filters = configUtils.getFilters(metric);
    if (_.isUndefined(filters) || _.isEmpty(filters)){
        return res.status(404).json({
            'info': 'Filters Not Found'
        });
    }
    return res.json(filters);
};

    Get filter title using filter ID and metric type

    exports.getFilterTitle = function(req,res){
    var filterID = parseInt(req.params.filterID),
        metricType = req.query.metricType;

    if (_.isUndefined(filterID) || _.isUndefined(metricType)){
        return res.send(400);
    }
    var filter = _.findWhere(configFilters,{filterID:filterID});
    if (!_.isUndefined(filter)){
        if (metricType == constants.METRIC_TYPE_RESOURCE){
            return res.json({
                filterTitle: filter.titleResource
            });
        }
        if (metricType == constants.METRIC_TYPE_EXPLORER){
            return res.json({
                filterTitle: filter.titleExplorer
            });
        }
    }else{
        return res.send(404);
    }
};
