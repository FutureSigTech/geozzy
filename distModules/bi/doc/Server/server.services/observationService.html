        var resource = metricRequest.resource;
        var event = metricRequest.event;

        var seconds = parseInt(metricRequest.seconds);

        var metricResource = db_metricResourceModel.metricResourceModel();
        metricResource._observation =  observation._id;
        metricResource.metricTime = metricTime;
        metricResource.resource = resource;
        metricResource.event = event;
        metricResource.section = section;
        metricResource.seconds = seconds;
        observation.metrics.push(metricResource);
    }

            console.log(err);
            return res.status(400).json({
                "info": "Invalid observation fields"
            });
        }else{
            return res.status(201).json({
                "ID": observation._id

        var observation = db_observationModel.observationExplorerModel();
    observation.user_ID = user_ID;
    observation.session_ID = session_ID;
    observation.device = device;
    observation.observationTime = new Date(observationTime);
    observation.language = language;

    for (var i = 0; i<metrics.length; i++){
        var metricRequest = metrics[i];

        var metricTime = new Date(metricRequest.metricTime);
        var explorer_ID = metricRequest.explorer_ID;
        var bounds = metricRequest.bounds;
        var filters = metricRequest.filters;

        var metricExploration = db_metricExplorationModel.metricExplorationModel();
        metricExploration._observation =  observation._id;
        metricExploration.metricTime = metricTime;
        metricExploration.explorer_ID = explorer_ID;
        metricExploration.bounds  = bounds;
        metricExploration.filters = filters;
        observation.metrics.push(metricExploration);
    }
