    Callback to validate resource observation JSON structure

    exports.validateResourceObservation = function(req,res,next){
    var user_ID = req.body.user_ID;
    var session_ID = req.body.session_ID;
    var device = req.body.device;
    var observationTime = req.body.observationTime;
    var language = req.body.language;
    var metrics = req.body.metrics;

    if (_.isUndefined(user_ID) || _.isUndefined(session_ID)|| _.isUndefined(device) ||
        _.isUndefined(observationTime) || _.isUndefined(language) ||
        _.isUndefined(metrics) || !(metrics instanceof Array)){
        return res.status(400).json({
            "info":"Invalid observation fields"
        });
    }

    observationTime = new Date(observationTime);
    if (isNaN(observationTime.getTime())){
        return res.status(400).json({
            "info":"Invalid observation time"
        });
    }

    for (var i = 0; i<metrics.length; i++){
        var metricRequest = metrics[i];
        var metricTime = metricRequest.metricTime;

        metricTime = new Date(metricTime);
        if (_.isNaN(metricTime.getTime())){
            return res.status(400).json({
                "info":"Invalid metric time"
            });
        }
        var resource = metricRequest.resource;
        var event = metricRequest.event;
        var section = metricRequest.section;
        var seconds = parseInt(metricRequest.seconds);

        if (_.isUndefined(resource) || _.isUndefined(event) || _.isUndefined(section) || _.isNaN(seconds) ){
            return res.status(400).json({
                "info":"Invalid metric fields"
            });
        }
    }
