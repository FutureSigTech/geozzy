    Executes a derived or basic query depending of the metric type

    exports.statsStrategy = function(req,res){
    var metric = req.statParams.metric;
    var groupBy = req.statParams.groupBy;
    var countryID = req.statParams.countryID;
    var chartType = req.statParams.chartType;
    var filters = req.body.filters;
    var order = req.body.order;

    if (!_.isUndefined(metric.derivedMetricIDs) && _.size(metric.derivedMetricIDs) > 0){
        getDerivedStats(metric,groupBy,countryID,chartType,filters,order,res);
    }else{
        getBasicStats(metric,groupBy,countryID,chartType,filters,order,res);
    }
};

            return res.status(500);

    Returns a promise with a basic stats query

    var getPromiseBasicStats = function (metric,groupBy,countryID,chartType,filters,order) {
    var deferred = q.defer();
    var pipelineStages = aggregationUtils.buildPipeline(metric,groupBy,filters,countryID,order);

    console.log('Metric:\n' + metric.metricName);
    console.log("Model:\n " + metric.model);
    console.log("Aggregation Pipeline:\n %j", pipelineStages);

    var model = db_result.getModel(metric.model);
    var aggregation = model.aggregate(pipelineStages);
    aggregation.options = {allowDiskUse: true};
    aggregation.exec(
            function (err,data){
                if (err){
                    deferred.reject(err);
                }else{
                    var payload = {
                        chartType: chartType,
                        data: data
                    };
                    if (!_.isUndefined(countryID)){
                        payload["countryID"] = countryID;
                    }
                    deferred.resolve(payload);
                }
            }
    );
    return deferred.promise;
};

    Executes severall queries in parallel in order to calculate derived stats

    var getDerivedStats = function(metric,groupBy,countryID,chartType,filters,order,res){

    var metrics = _.map(metric.derivedMetricIDs, function(metricID){
        return configUtils.getMetric(metricID);
    });
    var promises = _.map(metrics, function(metric){
        return getPromiseBasicStats(metric,groupBy,countryID,chartType,filters,order);
    });
  • ¶

    Executes the promises in parallel and waits for the response of all of them

        q.allSettled(promises)
        .then(function(resultsPromise) {
            var chartType = resultsPromise[0].value.chartType,
                results0 = resultsPromise[0].value.data,
                results1 = resultsPromise[1].value.data,
                results = [];

            _.each(results0,function(result0){
                var result1 = _.findWhere(results1,{group:result0.group});
                if (!_.isUndefined(result1)){
                    results.push({
                        data: (result0.data/result1.data),
                        group: result0.group
                    });
                }
            });
            return res.json({
                chartType: chartType,
                data: results
            });
        });
};
