    exports.validateStatsParams = function(req,res,next){
    var metricID = req.body.metricID;
    var groupByID = req.body.groupByID;
    var countryID = req.body.countryID;
    var filters = req.body.filters;
    var order = req.body.order;

    if (!validation.validStatisticParams(metricID,groupByID,filters,order)){
        return res.send(400);
    }
    groupByID = parseInt(groupByID);
    var metric = configUtils.getMetric(metricID);
    if (_.isUndefined(metric) || (_.isUndefined(metric.model) && _.isUndefined(metric.derivedMetricIDs))){
        return res.status(404).json({
            'info': 'Metric Not Found'
        });
    }
    var groupBy = configUtils.getGroupBy(groupByID);
    if (_.isUndefined(groupBy) || _.isUndefined(groupBy.group)){
        return res.status(404).json({
            'info': 'Organization criteria Not Found'
        });
    }
    var chartType = configUtils.getChartType(metric,groupBy);
    if (_.isUndefined(chartType)){
        return res.status(404).json({
            'info': 'Chart type Not Found'
        })
    }

    req.statParams = {
        metric: metric,
        groupBy: groupBy,
        countryID: countryID,
        chartType: chartType
    };

    next();
};
