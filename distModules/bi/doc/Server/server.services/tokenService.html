    Callback to generate an auth token

    exports.createToken = function(req,res){
    var session_ID = req.body.session_ID || '';

    if (session_ID == ''){
        return res.send(400);
    }

    var token = jwt.sign({session_ID:session_ID},secret.secretToken,{expiresInMinutes:configServer.tokenExpiration});
    return res.status(200).json({token:token});
};
