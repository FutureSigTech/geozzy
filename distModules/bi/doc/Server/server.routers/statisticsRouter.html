    Router for /statistics

  • ¶
    • express: express: NodeJS server library
    • router: express Router instance
    • configServer: server configuration module
    • statistics: stats service module
    • validateStatistics: stats validation middleware
    • jwt: JSON web token instance
    • secret: Used to generate new tokens
    var express = require('express'),
    router = express.Router(),
    configServer = require('../config/system/configServer'),
    statistics = require('../services/statisticsService'),
    validateStatistics = require('../services/validateStatisticsService'),
    jwt = require('express-jwt'),
    secret = require('../config/system/secret');

if (configServer.authToken){
    router
        .route('*')
        .all(jwt({secret: secret.secretToken}));
}

router
    .route('/')
    .post(validateStatistics.validateStatsParams,statistics.statsStrategy);

module.exports = router;
