    Schema fields:

    • user_ID: user ID
    • session_ID: session ID
    • device: info about the user device
    • observationTime: time in which the metric was sent
    • language: user language
    • metric: metric data
    • remoteIP: user IP address
    • remoteLocation: info about the user location
    var resultSchema = new mongoose.Schema({
    user_ID: { type: Number, required: true },
    session_ID: { type: String, required: true },
    device: { type: Object, required: true },
    observationTime: { type: Date, required: true },
    language: { type: String, required: true },
    metric: { type: Object, required: true },
    remoteIP: { type: String },
    remoteLocation: { type: Object }
});
exports.resultSchema = resultSchema;

    Get a model dynamically using the model name and the year

    exports.getModel = function(modelName){
    var year = (new Date()).getFullYear();
    var collectionName = modelName+"_"+year;
    var modelFound = _.findWhere(modelsCache,{collectionName: collectionName});
    if (_.isUndefined(modelFound)){
        model = mongoose.model(modelName, resultSchema, collectionName);
        modelsCache.push({
            collectionName: collectionName,
            model: model
        });
    }else{
        model = modelFound.model;
    }
    return model;
};
