        url: function () {
            var url = this.urlRoot + "/filter"
            if (!_.isUndefined(this.get('filterID'))) {
                url += "/" + this.get('filterID');
            }
            if (!_.isUndefined(this.get('metricType'))) {
                url += "?metricType=" + this.get('metricType');
            }
            return url;
        },
