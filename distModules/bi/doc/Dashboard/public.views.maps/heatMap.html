            renderHeatMapLayer: function () {
            var cfgHeatMap = {
                "radius": .005,
                "maxOpacity": .5,
                "scaleRadius": true,
                "useLocalExtrema": true,
                latField: 'lat',
                lngField: 'lng',
                valueField: 'data'
            };
            var heatMapLayer = new HeatmapOverlay(cfgHeatMap);
            this.map.addLayer(heatMapLayer);
            var heatMapData = {
                max: 50,
                data: this.data
            };
            heatMapLayer.setData(heatMapData);
        },
